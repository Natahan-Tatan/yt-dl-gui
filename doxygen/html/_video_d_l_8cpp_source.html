107 void* VideoDL::Process(void* p)
108 {
109  VideoDL* data = static_cast<VideoDL*>(p);
110 
111  std::string exe = GetPathToYoutubeDl();
112 
113  struct stat i;
114 
115  if(stat(exe.c_str(), &i) != 0)
116  {
117  Fl::lock();
118  fl_alert("The youtube-dl program must be download.");
119  if(!DownloadLastYoutubeDl())
120  {
121  return nullptr;
122  }
123  Fl::unlock();
124  }
125 
126  std::string cmd = exe + " " + data->_url + "--no-color --newline --output \"" + data->_outdir + "/%(title)s.%(ext)s\" ";
127 
128  if(data->_audioOnly)
129  {
130  cmd += " -x --audio-format=\"" + data->_audioType + "\"";
131  }
132  else
133  {
134  cmd += " -f best";
135  }
136 
137  std::array<char, 1024> buffer;
138  std::unique_ptr<FILE, decltype(&pclose)> pipe(popen(cmd.c_str(), "r"), pclose);
139 
140  while (fgets(buffer.data(), buffer.size(), pipe.get()) != nullptr)
141  {
142  std::cout << buffer.data();
143 
144  std::string str = buffer.data();
145  int firstSpace = str.find(" ");
146  int firstPercent = str.find("%");
147 
148  if(firstSpace != std::string::npos && firstPercent != std::string::npos)
149  {
150  std::string value = str.substr(firstSpace, firstPercent-firstSpace);
151 
152  if(!value.empty())
153  {
154  pthread_mutex_lock(&data->_mutex);
155  data->_progress = std::stof(value);
156  pthread_mutex_unlock(&data->_mutex);
157  }
158  }
159  else
160  {
161  if(str.find("ERROR") != std::string::npos || str.find("error") != std::string::npos)
162  {
163  fl_alert("Error: %s", str.c_str());
164  std::cout << "Failure of process" << std::endl;
165  return nullptr;
166  }
167  }
168 
169  }
170 
171  pthread_mutex_lock(&data->_mutex);
172  data->_progress = 101;
173  pthread_mutex_unlock(&data->_mutex);
174 
175  std::cout << "End of process" << std::endl;
176  return nullptr;
177 }
