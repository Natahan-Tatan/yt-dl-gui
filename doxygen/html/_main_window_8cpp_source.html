185 void MainWindow::UpdateProgress(void* u)
186 {
187  MainWindow* window = static_cast<MainWindow*>(u);
188  if(!window->_current_vdl)
189  return;
190 
191  float p = window->_current_vdl->GetProgress();
192  std::ostringstream oss;
193  oss << p << "%";
194 
195  if(p < 100)
196  {
197  Fl::lock();
198  window->progress->value(p);
199  window->progress->copy_label(oss.str().c_str());
200  Fl::unlock();
201  }
202  else if(p == 100)
203  {
204  Fl::lock();
205  window->progress->value(100);
206  window->progress->copy_label("Processing...");
207  Fl::unlock();
208  }
209  else if(p >= 101)
210  {
211  Fl::lock();
212  window->progress->copy_label("Ended !");
213  window->setup_gui->activate();
214  Fl::unlock();
215 
216  Fl::remove_idle(UpdateProgress, u);
217  }
218 }
